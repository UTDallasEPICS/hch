model AppForm {
    id          Int           @id @default(autoincrement())
    userId      String
    status      String
    submittedAt DateTime      @default(now())
    questions   AppQuestion[]
}

enum QuestionType {
    TEXT
    MULTIPLE_CHOICE
    CHECKLIST
    MULTI_PLUS_TEXT
}

    YOUR_EMAIL_ADDRESS
    YOUR_PHONE_NUMBER
    YOUR_GENDER
    YOUR_DATE_OF_BIRTH
    YOUR_ADDRESS

    CHILD_FIRST_NAME
    CHILD_LAST_NAME
    CHILD_DATE_OF_BIRTH
    CHILD_GENDER
    CHILD_ADDRESS

    MEDICAL_DIAGNOSIS
    MEDICAL_DIAGNOSIS_DATE
    HOUSEHOLD_MEMBERS

    BIO_PARENTS_RESIDE
    CUSTODY_HOLDER
    PRIMARY_CONTACT_MAIN

    LEGAL_MOTHER_FIRST_NAME
    LEGAL_MOTHER_LAST_NAME
    LEGAL_MOTHER_STREET
    LEGAL_MOTHER_CITY
    LEGAL_MOTHER_STATE
    LEGAL_MOTHER_ZIP
    LEGAL_MOTHER_EMAIL

    LEGAL_FATHER_FIRST_NAME

    SCHOLARSHIP_FOR_WHO
    THERAPIST_OR_REFERRAL
    MENTAL_HEALTH_INSURANCE
}

model AppQuestion {
    id           Int            @id @default(autoincrement())
    formId       Int
    questionKey  AppQuestionKey
    questionType QuestionType

    answerText   String?
    answerChoice String?
    answerList   String?

    form         AppForm        @relation(fields: [formId], references: [id])

    @@unique([formId, questionKey])
}
